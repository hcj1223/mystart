# 最长连续不重复子序列

https://www.acwing.com/problem/content/description/801/

```c++
#include <iostream>
using namespace std;
const int N = 1e5 + 7;
int q[N], s[N];
int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> q[i];
    int res = 0;
    for (int i = 0, j = 0; i < n; i++)
    {
        s[q[i]]++;
        while (s[q[i]] > 1)
        {
            s[q[j]]--;
            j++;
        }
        res = max(res, i - j + 1);
    }
    cout << res << endl;
    return 0;
}
```
