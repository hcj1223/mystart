# 单调栈

https://www.acwing.com/problem/content/832/

```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 7;
int n;
stack<int> stk;
int main()
{
    cin >> n;
    for (int i = 0; i < n; i++)
    {
        int x;
        cin >> x;
        while (stk.size() && stk.top() >= x) //保证栈中元素单调递增
            stk.pop();
        if (stk.size())
            cout << stk.top() << ' ';
        else
            cout << "-1" << ' ';
        stk.push(x);
    }
    return 0;
}
```
