# 求逆序对个数

```c++
long long qsort(int l, int r)
{
    if (l >= r)
        return 0;
    int mid = l + r >> 1;
    long long res = qsort(l, mid) + qsort(mid + 1, r);
    int i = l, j = mid + 1, k = 0;
    while (i <= mid && j <= r)
        if (a[i] <= a[j])
            t[k++] = a[i++];
        else
        {
            res += mid - i + 1;
            t[k++] = a[j++];
        }
    while (i <= mid)
        t[k++] = a[i++];
    while (j <= r)
        t[k++] = a[j++];
    for (int i = l, j = 0; i <= r; i++, j++)
        a[i] = t[j];
    return res;
}
```
