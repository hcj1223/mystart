# 判断负环

https://www.acwing.com/problem/content/description/854/

```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 7;
int n, m;
int h[N], e[N], ne[N], w[N], idx;
int dis[N];
bool st[N]; //存贮该元素是否在队列中
void add(int a, int b, int c)
{
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx++;
}
int cnt[N];
int spfa()
{
    memset(dis, 0x3f, sizeof h);
    dis[u] = 0;
    queue<int> q;
    for (int i = 1; i <= n; i++)
    {
        q.push(i);
        st[i] = 1;
    }
    while (q.size())
    {
        int t = q.front();
        q.pop();
        st[t] = 0;
        for (int i = h[t]; i != -1; i = ne[i])
        {
            int j = e[i];
            if (dis[j] > dis[t] + w[i])
            {
                dis[j] = dis[t] + w[i];
                cnt[j] = cnt[t] + 1;
                if (cnt[j] >= n)
                    return 1;
                if (!st[j])
                {
                    q.push(j);
                    st[j] = 1;
                }
            }
        }
    }
    return 0;
}
signed main()
{
    memset(h, -1, sizeof h);
    cin >> n >> m;
    while (m--)
    {
        int a, b, c;
        cin >> a >> b >> c;
        add(a, b, c);
    }
    if (spfa())
        puts("Yes");
    else
        puts("No");
    return 0;
}
```
