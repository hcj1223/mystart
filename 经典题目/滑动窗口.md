# 滑动窗口（单调队列）

https://www.acwing.com/problem/content/156/

```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e6 + 7;
int n, k, a[N];
deque<int> q;
int main()
{
    cin >> n >> k;
    for (int i = 0; i < n; i++)
        cin >> a[i];
    for (int i = 0; i < n; i++)
    {
        if (q.size() && i - k + 1 > q.front())
            q.pop_front();
        while (q.size() && a[q.back()] >= a[i])
            q.pop_back();
        q.push_back(i);
        if (i >= k - 1)
            cout << a[q.front()] << " ";
    }
    puts("");
    while (q.size())
        q.pop_back();
    for (int i = 0; i < n; i++)
    {
        if (q.size() && i - k + 1 > q.front())
            q.pop_front();
        while (q.size() && a[q.back()] <= a[i])
            q.pop_back();
        q.push_back(i);
        if (i >= k - 1)
            cout << a[q.front()] << " ";
    }
    return 0;
}
```
