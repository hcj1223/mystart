# 染色法

```c++
int color[N];
bool dfs(int u, int c)
{
    color[u] = c;
    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        if (!color[j])
        {
            if (!dfs(j, 3 - c))	//染成 2 或 1
                return 0;
        }
        else if (color[j] == c) // 矛盾
            return 0;
    }
    return 1;
}
bool isBgraph()
{
    for (int i = 1; i <= n; i++)
        if (!color[i])
            if (!dfs(i, 1))
                return 0;
    return 1;
}
```
