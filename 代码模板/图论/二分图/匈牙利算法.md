

# 匈牙利算法

```c++
int match[N]; //存储右边的点和左边哪个点匹配
bool f[N];
bool findmatch(int u)
{
    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        if (!f[j])
        {
            f[j] = 1;
            if (match[j] == 0 || findmatch(match[j])) // 如果右边点未匹配或左边点可以和其他点匹配
            {
                match[j] = u;
                return 1;
            }
        }
    }
    return 0;
}
int Maxmatch()
{
    int res = 0;
    for (int i = 1; i <= n1; i++) // 遍历左边的点
    {
        memset(f, 0, sizeof f);
        if (findmatch(i))
            res++;
    }
    return res;
}
```
