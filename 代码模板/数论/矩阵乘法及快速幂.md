# 矩阵快速幂

```c++
struct Matrix
{
    int g[N][N];
    void init()
    {
        memset(g, 0, sizeof g);
        for (int i = 0; i < n; i++)
            g[i][i] = 1;
    }
    friend Matrix operator*(Matrix a, Matrix b)
    {
        Matrix c;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
            {
                c.g[i][j] = 0;
                for (int k = 0; k < n; k++)
                    c.g[i][j] += a.g[i][k] * b.g[k][j] % mod;
            }
        return c;
    }
    friend Matrix operator^(Matrix a, int k)
    {
        Matrix res;
        res.init();
        while (k)
        {
            if (k & 1)
                res = res * a;
            a = a * a;
            k >>= 1;
        }
        return res;
    }
    void print()
    {
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
                cout << g[i][j] << " ";
            
        }
    }
};
```
