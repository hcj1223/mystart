# 中国剩余定理

![中国剩余定理](https://user-images.githubusercontent.com/95573252/206192619-3827244f-456a-4349-b3ae-eb59edd7c890.png)

#### 证明过程：https://www.acwing.com/solution/content/21747/

```c++
void solve()
{
    ll a1, m1;
    cin >> a1 >> m1;
    bool f = 1;
    while (--n)
    {
        ll a2, m2;
        cin >> a2 >> m2;
        ll k1, k2;
        ll d = exgcd(a1, a2, k1, k2);
        if ((m2 - m1) % d)
        {
            f = 0;
            break;
        }
        k1 *= (m2 - m1) / d;
        ll t = a2 / d;
        k1 = (k1 % t + t) % t; //使k1为正数
        m1 += a1 * k1;
        a1 = abs(a1 / d * a2);
    }
    if (!f) //无解
        cout << -1 << '\n';
    else
        cout << (m1 % a1 + a1) % a1 << '\n';
}
```
