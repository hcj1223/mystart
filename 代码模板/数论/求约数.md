# 1.求所有约数

```c++
vector<int> getdivide(int n)
{
    vector<int> res;
    for (int i = 1; i <= n / i; i++)
        if (n % i == 0)
        {
            res.push_back(i);
            if (i != n / i)
                res.push_back(n / i);
        }
    sort(res.begin(), res.end());
    return res;
}
```

# 2.求质约数

```c++
unordered_map<int, int> getdivide(int n)
{
    unordered_map<int, int> primes;
    for (int i = 2; i <= n / i; i++)
        while (n % i == 0)
        {
            n /= i;
            primes[i]++;
        }
    if (n > 1)
        primes[n]++;
}
```
