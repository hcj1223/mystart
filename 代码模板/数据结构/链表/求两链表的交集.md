# 求两链表的交集

```c++
list getsame(list &a, list &b)
{
    list c;
    c = new node;
    c->next = NULL;
    list pa = a->next, pb = b->next, pc = c;
    while (pa && pb)
    {
        if (pa->data < pb->data)
            pa = pa->next;
        else if (pa->data > pb->data)
            pb = pb->next;
        else
        {
            pc->next = pa;
            pc = pa;
            pa = pa->next;
            pb = pb->next;
        }
    }
    return c;
}
```
