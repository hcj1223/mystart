# 求哈夫曼编码

```c++
vector<string> huffmansolve(huffmantree &t, int len)
{
    vector<string> ans;
    for (int i = 0; i < len; i++)
    {
        string str;
        int child = i;
        int parent = t.nodes[i].parent;
        do
        {
            if (t.nodes[parent].lchild == child)
                str = '0' + str;
            else
                str = '1' + str;
            child = parent;
            parent = t.nodes[child].parent;
        } while (parent != -1);
        ans.push_back(str);
    }
    return ans;
}
```
