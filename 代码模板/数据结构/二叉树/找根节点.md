# 二叉树找根节点

```c++
int findroot()
{
    vector<int> v;
    for (int i = 0; i < n; i++)
    {
        if (lch[i])
            v.push_back(lch[i]);
        if (rch[i])
            v.push_back(rch[i]);
    }
    for (int i = 0; i < n; i++)
        if (find(v.begin(), v.end(), i) == v.end())
            return i;
}
```
