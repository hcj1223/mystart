# 并查集

```c++
int p[N], size[N]; // p[]存放每个点的父节点,size[]存放每个集合的点的个数
int d[N];          // 带权并查集
//找到x的祖宗节点
int find(int x)
{
    if (p[x] != x)
    {
        int t = p[x];
        p[x] = find(p[x]); //路径压缩
        d[x] += d[t];
    }
    return p[x];
}
//初始化
void init()
{
    for (int i = 1; i <= n; i++)
        p[i] = i, size[i] = 1, d[i] = 0;
}
//两个集合合并成树
void merge(int x, int y, int w)
{
    p[find(x)] = find(y);
    size[find(y)] += size[find(x)];
    d[find(x)] = d[find(y) - d(find(x))] + w;
}
//查询两个元素是否在一个集合里
bool check(int x, int y)
{
    return find(x) == find(y);
}
```
