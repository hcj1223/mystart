# 树状数组

```c++
int tree[N];
int lowbit(int x)
{
    return x & -x;
}
void update(int x, int d) // 单点修改:a[x] += d
{
    while (x <= N)
    {
        tree[x] += d;
        x += lowbit(x);
    }
}
int sum(int x) // 求区间和s[1..x]
{
    int ans = 0;
    while (x > 0)
    {
        ans += tree[x];
        x -= lowbit(x);
    }
    return ans;
}
```
