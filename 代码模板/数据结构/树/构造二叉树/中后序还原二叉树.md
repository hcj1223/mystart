# 中后序还原二叉树

```c++
tree createtree(char *suc, char *mid, int n)
{
    if (n <= 0)
        return NULL;
    tree t = new treenode;
    t->data = *(suc + n - 1);
    int rootpos = -1;
    for (int i = 0; i < n; i++)
        if (mid[i] == *(suc + n - 1))
        {
            rootpos = i;
            break;
        }
    t->lchild = createtree(suc, mid, rootpos);
    t->rchild = createtree(suc + rootpos, mid + rootpos + 1, n - rootpos - 1);
    return t;
}
```
