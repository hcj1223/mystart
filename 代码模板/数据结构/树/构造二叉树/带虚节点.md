char *createtree(tree &t, char *str)
{
    if (*str == '#')
    {
        t = NULL;
        return str + 1;
    }
    t = new treenode;
    t->data = *str;
    char *strafterleft = createtree(t->lchild, str + 1);
    char *strafterright = createtree(t->rchild, strafterleft);
    retturn strafterright;
}
