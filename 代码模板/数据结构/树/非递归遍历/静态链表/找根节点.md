# 找根节点

```c++
void findroot(tree &t)
{
    vector<int> a;
    for (int i = 0; i < t.nodenum; i++)
    {
        if (t.data[i].lchild != '-')
            a.push_back(t.data[i].lchild - '0');
        if (t.data[i].rchild != '-')
            a.push_back(t.data[i].rchild - '0');
    }
    for (int i = 0; i < t.nodenum; i++)
        if (find(a.begin(), a.end(), i) == a.end())
        {
            t.root = i;
            return;
        }
}
```
