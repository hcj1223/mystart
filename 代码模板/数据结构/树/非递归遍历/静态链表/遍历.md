# 遍历

## 1.前序

```c++
string preprint(tree &t)
{
    if (t.nodenum <= 0)
        return "";
    string str;
    stack<treenode> stk;
    stk.push(t.data[t.root]);
    while (stk.size())
    {
        auto p = stk.top();
        stk.pop();
        str.push_back(p.data);
        if (p.rchild != '-')
            stk.push(t.data[p.rchild - '0']);
        if (p.lchild != '-')
            stk.push(t.data[p.lchild - '0']);
    }
    return str;
}
```

## 2.中序

```c++
string midprint(tree &t)
{
    if (t.nodenum <= 0)
        return "";
    string str;
    stack<treenode> stk;
    stk.push(t.data[t.root]);
    while (stk.size())
    {
        if (stk.top().occurtimes == 0)
        {
            stk.top().occurtimes++;
            if (stk.top().lchild != '-')
                stk.push(t.data[stk.top().lchild - '0']);
        }
        else
        {
            auto p = stk.top();
            stk.pop();
            str.push_back(p.data);
            if (p.rchild != '-')
                stk.push(t.data[p.rchild - '0']);
        }
    }
    return str;
}
```

## 3.后序

```c++
string sucprint(tree &t)
{
    if (t.nodenum <= 0)
        return "";
    string str;
    stack<treenode> stk;
    stk.push(t.data[t.root]);
    while (stk.size())
    {
        if (stk.top().occurtimes == 0)
        {
            stk.top().occurtimes++;
            if (stk.top().lchild != '-')
                stk.push(t.data[stk.top().lchild - '0']);
        }
        else if (stk.top().occurtimes == 1)
        {
            stk.top().occurtimes++;
            if (stk.top().rchild != '-')
                stk.push(t.data[stk.top().rchild - '0']);
        }
        else
        {
            str.push_back(stk.top().data);
            stk.pop();
        }
    }
    return str;
}
```
