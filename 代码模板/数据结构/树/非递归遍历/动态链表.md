## 3.后序

```c++
string sucprint(tree &t)
{
    stack<tree> stk;
    string res;
    tree p = t, q = NULL;
    while (p || stk.size())
    {
        while (p)
        {
            stk.push(p);
            p = p->lchild;
        }
        p = stk.top();
        stk.pop();
        if (p->rchild == NULL || p->rchild == q)
        {
            q = p;
            res.push_back(p->data);
            p = NULL;
        }
        else
        {
            stk.push(p);
            p = p->rchild;
        }
    }
    return res;
}
```
