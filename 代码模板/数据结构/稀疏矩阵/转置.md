# 转置

```c++
typedef pair<pair<int, int>, int> pll;
vector<pll> transmatrix(vector<pll> &a, int n)
{
    int m = a.size();
    vector<pll> b(m);
    vector<int> rpos(n);
    vector<int> rsum(n, 0);
    for (int i = 0; i < a.size(); i++)
        rsum[a[i].first.second]++;
    rpos[0] = 0;
    for (int i = 1; i < n; i++)
        rpos[i] = rpos[i - 1] + rsum[i - 1];
    for (int i = 0; i < a.size(); i++)
    {
        int l = a[i].first.first, r = a[i].first.second, w = a[i].second;
        int j = rpos[r];
        b[j] = {{r, l}, w};
        rpos[r]++;
    }
    return b;
}
```
