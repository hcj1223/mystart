```c++
void transmatrix(trimatrix &a, trimatrix &b)
{
    b.mu = a.nu, b.nu = a.mu, b.tu = a.tu;
    b.datas = new trinode[b.tu];
    int *rsum = new int[b.mu];
    for (int i = 0; i < b.mu; i++)
        rsum[i] = 0;
    for (int i = 0; i < a.tu; i++)
        rsum[a.datas[i].col]++;
    int *rpos = new int[b.mu];
    rpos[0] = 0;
    for (int i = 1; i < b.mu; i++)
        rpos[i] = rpos[i - 1] + rsum[i - 1];
    for (int i = 0; i < a.tu; i++)
    {
        int j = rpos[a.datas[i].col];
        b.datas[j].row = a.datas[i].col;
        b.datas[j].col = a.datas[i].row;
        b.datas[j].data = a.datas[i].data;
        rpos[a.dats[i].col]++;
    }
    delete[] rsum;
    delete[] rpos;
}
```

```
typedef pair<pair<int, int>, int> pll;
vector<pll> transmatrix(vector<pll> &a, int n)
{
    int m = a.size();
    vector<pll> b(m);
    vector<int> rpos(n);
    vector<int> rsum(n, 0);
    for (int i = 0; i < a.size(); i++)
        rsum[a[i].first.second]++;
    rpos[0] = 0;
    for (int i = 1; i < n; i++)
        rpos[i] = rpos[i - 1] + rsum[i - 1];
    for (int i = 0; i < a.size(); i++)
    {
        int l = a[i].first.first, r = a[i].first.second, w = a[i].second;
        int j = rpos[r];
        b[j] = {{r, l}, w};
        rpos[r]++;
    }
    return b;
}
```
