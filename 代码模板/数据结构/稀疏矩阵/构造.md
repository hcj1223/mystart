# 构造

```c++
struct trinode
{
    int row, col, data;
};
struct trimatrix
{
    trinode *datas;
    int mu, nu, tu;
};
void initrimatrix(trimatrix &t, int num[], int m, int n)
{
    t.mu = m, t.nu = n;
    int cnt = 0;
    for (int i = 0; i < m * n; i++)
        if (num[i])
            cnt++;
    t.tu = cnt;
    t.datas = new trinode[cnt];
    int pos = 0, k = 0;
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
        {
            if (num[pos])
            {
                t.datas[k].row = i, t.datas[k].col = j, t.datas[k].data = num[pos];
                k++;
            }
            pos++;
        }
}
```
