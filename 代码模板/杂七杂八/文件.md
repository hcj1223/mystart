## 文件

```c++
#include <stdio.h>
int main()
{
    FILE *fp;
    fp = fopen("file.txt", "r");
    char c = fgetc(fp); //从文件中读取字符

    char s = 's';
    fputc(s, fp); //将字符写入文件

    int n;
    char str1;
    fgets(str1, n, fp); //从文件中读取含n个字符的字符串，保存到str1中

    char str2 = "hello";
    fputs(str2, fp); //将字符串str2写入文件

    int i = 0;
    float f = 0.0;
    char cc = 'c';
    fscanf(fp, "%d%f%c", &i, &f, &cc); //按指定格式读取数据
    fscanf(fp, "%d%f%c", i, f, cc);    //按指定格式写入数据

    fclose(fp);
}
```
